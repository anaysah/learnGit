git --a     //add all files in stage area and display in green color 
git add <filename> // you can add any file to stage area
git commit -m <"comment"> //you can commit all the files of stage area
git log // you can get all commit and branches
rm -rf .git to delete the repository

git add . //to add all files

to ignore files 
	create a file named <.gitignore>
	add all the file name in <.gitignore>
	you can also ignore directory by adding its name in <.gitignore>
	
	dir/ for all folder with name dir
	/dir/ for only outer
	*.txt for all txt files

showing changes between commits
	git diff // to compare stage area with working area
	git diff --staged // to compare stage are with previous commit

How to skip stage area
	git commit -a -m "comment"

How to remove and rename
	git rm <filename> // to remove the file
	git mv <oldname> <newname>  // to rename the files

How to remove file for tracking area
	git rm --cached <filename> // this will stop tracking this file

How to see changes and commit
	git log -p // this will show all commits in details
	git log -p -2 // this will show only 2 commits
	git log -stat // this will show commit in summary
	git log --pretty=oneline // all details in oneliner
	git log --pretty=full // more info but in readeable look
	git log --pretty=format:"<format>" // this will format the data and display as your requirements
	git commit --amend // to redit the arealdy commit
	git commit --amend --no-edit // to redit without changing any thing, used for add files

How to get back all data of file
	git checkout -- <filename>
