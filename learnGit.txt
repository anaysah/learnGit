git
git config --global user.name "ajay"
git config --global user.email "email@email.com"
git status
git config --list
git --a     //add all files in stage area and display in green color 
git add <filename> // you can add any file to stage area
git commit -m <"comment"> //you can commit all the files of stage area
git log // you can get all commit and branches
rm -rf .git to delete the repository

git add . //to add all files

to ignore files 
	create a file named <.gitignore>
	add all the file name in <.gitignore>
	you can also ignore directory by adding its name in <.gitignore>
	
	dir/ for all folder with name dir
	/dir/ for only outer
	*.txt for all txt files

showing changes between commits
	git diff // to compare stage area with working area
	git diff --staged // to compare stage are with previous commit
